<?php
    // Suppress Postgres error messages
    error_reporting(error_reporting() & 0xFFFD);

    // connect to the database 
    $conn = pg_connect("localhost", "8888", "bsmith");
    if ($conn == "") { 
        printf("$s%s%s", "Unable to connect to application.<br>",
            "Please verify that the application is running and ",
            "listening on port 8888.<br>");
        exit();
    }

    // Headings
    print("<table border=3 cellpadding=4 align=center width=85%>\n");
    print("<tr><th>MyInt</th>\n");
    print("    <th>MyFloat</th>\n");
    print("    <th>Notes</th>\n");
    print("</tr>\n");

    // execute query 
    $command = "SELECT myint, myfloat, notes FROM mytable";
    $result = pg_exec($conn, $command);
    if ($result == "") { 
        print("<p><font color=\"red\" size=+1>SQL Command failed!</p>");
        print("<p>Command: $command</p>\n");
        exit();
    }
    for($row = 0; $row < pg_NumRows($result); $row++)
    {
        $myintstr  = pg_result($result, $row, 0);
        $myfloatstr  = pg_result($result, $row, 1);
        $mynotes  = pg_result($result, $row, 2);
        print("<tr>\n");
        print("<td>$myintstr</td>\n");
        print("<td>$myfloatstr</td>\n");
        print("<td>$mynotes</td>\n");
        print("</tr>\n");
    }

    print("</table>\n");

    // free the result and close the connection 
    pg_freeresult($result);
    pg_close($conn);
?>
